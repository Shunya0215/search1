    function wrapChartLabels(label, maxWidthChars = 16) {

        if (typeof label !== 'string') return label; // Return as is if not a string

        if (label.length <= maxWidthChars) return label;


        const words = label.split(''); // Split by characters for Japanese

        let lines = [];

        let currentLine = "";

        for (let i = 0; i < words.length; i++) {

            if (currentLine.length + words[i].length <= maxWidthChars) {

                currentLine += words[i];

            } else {

                lines.push(currentLine);

                currentLine = words[i];

            }

        }

        lines.push(currentLine);

        return lines;

    }

    const optimizedAdOutput = document.getElementById('optimizedAdOutput');

    const loadingSpinner = document.getElementById('loadingSpinner');


    optimizeAdButton.addEventListener('click', async () => {

        const jobDescription = jobDescriptionInput.value.trim();

        let selectedTargetAudience = '';

        targetAudienceRadios.forEach(radio => {

            if (radio.checked) {

                selectedTargetAudience = radio.value;

            }

        });


        if (!jobDescription) {

            optimizedAdOutput.classList.remove('hidden');

            optimizedAdOutput.textContent = '求人内容を入力してください。';

            optimizedAdOutput.style.borderColor = energeticPlayfulPalette.red;

            return;

        }

        loadingSpinner.style.display = 'block';

            let chatHistory = [];

            const prompt = `あなたはデリバリー九州社の採用担当者です。以下の求人内容を、指定されたターゲット層に最も響くように最適化してください。

            最適化された広告文を、箇条書きや強調を使い、魅力的に作成してください。`;


            chatHistory.push({ role: "user", parts: [{ text: prompt }] });

            const payload = { contents: chatHistory };

            const apiKey = ""; 

            const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


            const response = await fetch(apiUrl, {

                method: 'POST',

                headers: { 'Content-Type': 'application/json' },

                body: JSON.stringify(payload)

            });


            const result = await response.json();

            

            if (result.candidates && result.candidates.length > 0 &&

                result.candidates[0].content && result.candidates[0].content.parts &&

                result.candidates[0].content.parts.length > 0) {

                const text = result.candidates[0].content.parts[0].text;

                optimizedAdOutput.textContent = text;

                optimizedAdOutput.classList.remove('hidden');

            } else {

                optimizedAdOutput.textContent = '広告文の生成に失敗しました。もう一度お試しください。';

                optimizedAdOutput.classList.remove('hidden');

            }

        } catch (error) {

            console.error('Error calling Gemini API:', error);

            optimizedAdOutput.textContent = 'エラーが発生しました。ネットワーク接続を確認し、もう一度お試しください。';

            optimizedAdOutput.style.borderColor = energeticPlayfulPalette.red;

            optimizedAdOutput.classList.remove('hidden');

        } finally {

            loadingSpinner.style.display = 'none';

        }

    });
